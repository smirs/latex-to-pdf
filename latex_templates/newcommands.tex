  \num{\fpeval{round(\Rent/30,2)}}%
}

  \num{\fpeval{round(\PetDeposit * \NumPets, 2)}}%
}

\newcount\CommencementCount
\newcount\StartDateCount
\newcount\PreLeaseDuration

\newcommand{\ComputePreLeaseDays}{%
  \pgfcalendardatetojulian{\Commencement}{\CommencementCount}%
  \pgfcalendardatetojulian{\LeaseStart}{\StartDateCount}%
  \PreLeaseDuration=\StartDateCount
  \advance\PreLeaseDuration by -\CommencementCount
}

  }%
}
